# Calculate the first 10,000 terms of the series 1 - 1/3 + 1/5 - 1/7 + ... and multiply by 4
def calculate_series(terms):
    total = 0
    for n in range(terms):
        if n % 2 == 0:  # even index
            total += 1 / (2 * n + 1)
        else:  # odd index
            total -= 1 / (2 * n + 1)
    return total * 4

# Number of terms
num_terms = 10000
result = calculate_series(num_terms)
print(result)
